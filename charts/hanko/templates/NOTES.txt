{{- $fullName := include "hanko.fullname" . -}}
{{- $publicPort := .Values.service.ports.public.port -}}
{{- $adminPort := .Values.service.ports.admin.port -}}

1. Get the application URLs by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts.public }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- range $host := .Values.ingress.hosts.admin }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ $fullName }}-public)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ $fullName }}-public'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ $fullName }}-public --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ $publicPort }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "hanko.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8000 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8000:$CONTAINER_PORT
{{- end }}

2. Hanko is now installed! To get started:

   a) Set up your database:
{{- if .Values.postgresql.enabled }}
      PostgreSQL is installed as part of this release.
{{- else }}
      Make sure you have configured an external PostgreSQL database in values.yaml
{{- end }}

   b) Configure your application:
      - Public API:  http://{{ $fullName }}-public.{{ .Release.Namespace }}.svc.cluster.local:{{ $publicPort }}
      - Admin API:   http://{{ $fullName }}-admin.{{ .Release.Namespace }}.svc.cluster.local:{{ $adminPort }}

   c) Update CORS origins in config.server.public.cors.allowOrigins
      to include your frontend application URL

   d) Configure WebAuthn relying party origins in config.webauthn.relyingParty.origins
      to match your application domain

   e) For production deployments:
      - Set config.session.secret to a secure random value
      - Enable TLS by setting config.session.cookie.secure to true
      - Use external secrets for sensitive values (passwords, keys)
      - Enable ingress with TLS termination
